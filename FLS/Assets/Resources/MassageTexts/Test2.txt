SELECT "ビッグボーイ！！" 
	MESS "ビッグボーイ！！！！！" ショーン

	BRANCH "[1] == 1"
		MESS "せっかくだから！俺は左の道を選ぶぜ！！" JB
	ELSEBRANCH
		MESS "せっかくだから！俺は右の道を選ぶぜ！！" JB
	ENDBRANCH

SEL_ELSE "ビッグボーイだ！"
	MESS "あ！！！ビッグボーイだ！！！！" ショーン

	BRANCH "[1] == 1"
		MESS "うるさっ" まだら牛
	ENDBRANCH

SEL_ELSE "ビッグボーイがあるんだ！"
	MESS "あ！！！ビッグボーイがあるんだ！！！！" ショーン

	MESS "山々の神々が散々で峰々のサボテンです" JB
	SELECT "肯定する"
		MESS "あれがサボテンかぁ" ディズム
	SEL_ELSE "否定する"
		MESS "違います" まだら牛
	SEL_END

SEL_END
