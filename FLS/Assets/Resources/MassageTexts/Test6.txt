#============================================
#↓内約
#============================================

#ダイアログを設置する
#DIALOG_SET 管理番号 登録済のダイアログ番号 (x座漂 y座漂)
DIALOG_SET 1 0 30 30

#ダイアログに使う変数を初期化する
VALUE_SET 121 0

#DIALOG_WAIT 管理番号 条件式
DIALOG_WAIT "[121] != 0"

DIALOG_REMOVE 1
BRANCH "[121] == 1"
	MESS "はいを選択"
ENDBRANCH
BRANCH "[121] == 2"
	MESS "いいえを選択"
ENDBRANCH

MESS "ダイアログ終了"
#EVENTEND
